JPEGView.ini 设置

; 设置为 true 以使用高质量采样作为默认值。true开/false关 
HighQualityResampling=false

; 抗锯齿
; DownSamplingFilter 可以是 BestQuality、NoAliasing 或 Narrow
; BestQuality 滤镜会产生非常少量的混叠。
; NoAliasing 滤镜是一个 Lanczos 滤镜，当锐化设置为零时几乎不会产生混叠。
; Narrow 滤镜会产生相当多的混叠，但会极大地锐化，也会锐化 100% 的图像。
DownSamplingFilter=Narrow

; 窗口模式false/全屏模式true
ShowFullScreen=false 

; 边框模式false/无边框模式true
WindowBorderlessOnStartup=false

image/auto
DefaultWindowRect=auto

; 将鼠标移动到屏幕/窗口底部时，设置为true以显示底部面板。底板提供基本图像处理功能
ShowBottomPanel=false

; 设置为false以禁用在缩放和平移期间混合的缩略图图像
ShowZoomNavigator=false

; 不显示导航面板
ShowNavPanel=FALSE

; 在窗口模式下设最大
DefaultWindowRect=max
